# 3.7 IaC — Infrastructure as Code

## Overview

Infrastructure as Code (IaC) = ** declarative configuration management** — toàn bộ infrastructure được mô tả bằng code, versioned trong Git, và apply tự động.

```
┌─────────────────────────────────────────────────────────────┐
│  IaC Workflow                                                │
│                                                              │
│  1. WRITE  ──► code (Terraform / Pulumi / Ansible)           │
│                                                              │
│  2. PLAN   ──► `terraform plan` → preview changes           │
│              (dry-run, xem diff trước khi apply)            │
│                                                              │
│  3. APPLY  ──► `terraform apply` → tạo/update resources      │
│              (idempotent — chạy lại cho cùng result)        │
│                                                              │
│  4. STATE  ──► terraform.tfstate (desired state)           │
│              (single source of truth)                       │
└─────────────────────────────────────────────────────────────┘
```

### Imperative vs Declarative

| | Imperative | Declarative |
|---|---|---|
| **Cách tiếp cận** | Step-by-step commands | Define desired state |
| **Ví dụ** | Ansible | Terraform |
| **Phong cách** | "Làm X rồi làm Y" | "Trạng thái mong muốn là Z" |
| **Tái hiện** | Script phải chạy đúng thứ tự | Config duy nhất |
| **Drift** | Script cũ không reflect infra hiện tại | ✅ State tracking |

## Terraform

### Core Concepts

```
┌──────────────────────────────────────────────────────────────┐
│  Terraform Workflow                                          │
│                                                              │
│  ┌──────────────┐                                            │
│  │  .tf files   │  (desired state: declarative)             │
│  │  main.tf     │                                            │
│  │  vars.tf     │                                            │
│  │  outputs.tf  │                                            │
│  └──────┬───────┘                                            │
│         │                                                     │
│  terraform init  (provider + module download)                │
│         │                                                     │
│  terraform plan  (dry-run: what's different?)               │
│         │                                                     │
│  terraform apply (make reality = desired state)              │
│         │                                                     │
│  ┌──────▼───────┐                                            │
│  │ terraform    │                                            │
│  │ .tfstate     │  (current state — local hoặc remote)      │
│  └──────────────┘                                            │
└──────────────────────────────────────────────────────────────┘
```

### Basic Structure

```hcl
# ─── main.tf ────────────────────────────────────────────────

terraform {
  required_version = ">= 1.7.0"

  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 5.0"
    }
  }

  # Remote state (best practice)
  backend "s3" {
    bucket         = "myorg-terraform-state"
    key            = "production/network/terraform.tfstate"
    region         = "ap-southeast-1"
    encrypt        = true
    dynamodb_table = "myorg-terraform-locks"  # State locking
  }
}

# Provider
provider "aws" {
  region = "ap-southeast-1"

  default_tags {
    tags = {
      Environment = "production"
      ManagedBy   = "terraform"
      Owner       = "devops-team"
    }
  }
}

# ─── Network ─────────────────────────────────────────────────

resource "aws_vpc" "main" {
  cidr_block           = "10.0.0.0/16"
  enable_dns_hostnames = true
  enable_dns_support   = true

  tags = {
    Name = "production-vpc"
  }
}

resource "aws_subnet" "private" {
  count             = 3  # 3 private subnets (multi-AZ)
  cidr_block        = cidrsubnet(aws_vpc.main.cidr_block, 8, count.index)
  vpc_id            = aws_vpc.main.id
  availability_zone = data.aws_availability_zones.available.names[count.index]

  tags = {
    Name = "private-subnet-${count.index + 1}"
    Type = "private"
  }
}

resource "aws_subnet" "public" {
  count                   = 3
  cidr_block              = cidrsubnet(aws_vpc.main.cidr_block, 8, count.index + 3)
  vpc_id                  = aws_vpc.main.id
  availability_zone       = data.aws_availability_zones.available.names[count.index]
  map_public_ip_on_launch = true  # NAT Gateway in public subnet

  tags = {
    Name = "public-subnet-${count.index + 1}"
    Type = "public"
  }
}

# ─── EKS Cluster ──────────────────────────────────────────────

resource "aws_eks_cluster" "main" {
  name     = "myapp-prod"
  role_arn = aws_iam_role.eks_cluster.arn
  version  = "1.29"

  vpc_config {
    subnet_ids              = aws_subnet.private[*].id
    endpoint_private_access = true
    endpoint_public_access  = true
  }

  kubernetes_network_config {
    ip_family         = "ipv4"
    service_ipv4_cidr = "172.16.0.0/16"
  }

  depends_on = [
    aws_iam_role_policy_attachment.eks_cluster_policy
  ]
}

# Node Group (managed)
resource "aws_eks_node_group" "main" {
  cluster_name    = aws_eks_cluster.main.name
  node_group_name = "production-nodes"
  node_role_arn   = aws_iam_role.eks_nodes.arn
  subnet_ids      = aws_subnet.private[*].id
  instance_types = ["t3.medium"]

  scaling_config {
    desired_size = 3
    min_size     = 2
    max_size     = 10
  }

  labels = {
    role = "worker"
  }

  taint {
    key    = "workload"
    value = "production"
    effect = "NO_SCHEDULE"
  }

  update_config {
    max_unavailable = 1
  }
}
```

### Dynamic Configuration

```hcl
# Dynamic blocks — khi không biết trước số lượng
resource "aws_security_group_rule" "ingress" {
  for_each = toset(["80", "443", "8080"])

  type              = "ingress"
  from_port         = each.value
  to_port           = each.value
  protocol          = "tcp"
  source_security_group_id = aws_security_group.app.id
  description = "Port ${each.value}"
}

# Dynamic expressions
locals {
  common_tags = merge(
    var.tags,
    {
      Environment = var.environment
      ManagedBy   = "terraform"
    }
  )
}

# for_each over map
resource "aws_instance" "app" {
  for_each = var.instances  # var.instances = { "web1": {...}, "web2": {...} }

  ami           = each.value.ami
  instance_type = each.value.type
  subnet_id     = each.value.subnet_id

  tags = merge(local.common_tags, {
    Name = each.key
  })
}

# Conditional resource
resource "aws_db_instance" "main" {
  # Only create RDS if enabled
  count = var.enable_rds ? 1 : 0

  identifier     = "myapp-db"
  engine         = "postgres"
  engine_version = "16.1"
  instance_class = var.db_instance_class

  # ...
}
```

### Terraform Functions

```hcl
locals {
  # String
  name = lower(replace(var.environment, "-", "_"))

  # Numeric
  subnet_count = length(var.availability_zones)

  # Collection
  private_subnets = [for s in aws_subnet.private : s.id]
  subnet_cidrs    = [for i, az in var.availability_zones :
    cidrsubnet(var.vpc_cidr, 8, i)
  ]

  # Object/Map transformation
  instance_tags = {
    for name, config in var.instances :
    name => merge(config, { name = name })
  }

  # Filtering
  active_nodes = [
    for node in aws_instance.node :
    node if node.arn != ""
  ]

  # Condition
  dynamo_config = var.enable_dynamo ? {
    billing_mode = "PAY_PER_REQUEST"
  } : {}
}
```

## Pulumi

### Why Pulumi? (vs Terraform)

| | Terraform (HCL) | Pulumi (Code) |
|---|---|---|
| **Language** | HCL (domain-specific) | TypeScript, Python, Go, C# |
| **Looping/Conditionals** | Limited (for_each, count, if) | Full language (map, filter, etc.) |
| **Reusability** | Modules + Terragrunt | Functions, classes |
| **IDE Support** | Limited | ✅ Full (autocomplete, type check) |
| **Testing** | Limited | ✅ Unit tests (Jest, pytest) |
| **Secrets** | External secrets with Provider | ✅ Native secret types |
| **Learning curve** | HCL specific | Use your known language |

### Pulumi Structure

```typescript
// ─── index.ts ────────────────────────────────────────────────

import * as pulumi from "@pulumi/pulumi";
import * as aws from "@pulumi/aws";

const config = new pulumi.Config();

// ─── VPC ──────────────────────────────────────────────────────

const vpc = new aws.ec2.Vpc("main", {
  cidrBlock: "10.0.0.0/16",
  enableDnsHostnames: true,
  enableDnsSupport: true,
  tags: {
    Name: `${config.require("environment")}-vpc`,
    ManagedBy: "pulumi",
  },
});

const privateSubnets = Array.from({ length: 3 }, (_, i) =>
  new aws.ec2.Subnet(`private-${i}`, {
    vpcId: vpc.id,
    cidrBlock: pulumi.interpolate`10.0.${i}.0/24`,
    availabilityZone: pulumi.Output.create(`ap-southeast-${["1a", "1b", "1c"][i]}`),
    tags: { Name: `private-subnet-${i}`, Type: "private" },
  })
);

// ─── EKS ──────────────────────────────────────────────────────

const cluster = new aws.eks.Cluster("main", {
  name: `myapp-${config.require("environment")}`,
  version: "1.29",
  roleArn: clusterRole.arn,
  vpcConfig: {
    subnetIds: privateSubnets.map((s) => s.id),
    endpointPrivateAccess: true,
    endpointPublicAccess: true,
  },
});

// ─── Outputs ──────────────────────────────────────────────────

export const vpcId = vpc.id;
export const eksEndpoint = cluster.endpoint;
export const kubeconfig = cluster.kubeconfig;
```

### Pulumi Testing

```typescript
// ─── __tests__/infrastructure.test.ts ─────────────────────────

import * as pulumi from "@pulumi/pulumi";
import * as aws from "@pulumi/aws";
import { describe, it, expect } from "@jest/globals";

// Mock all AWS resources for unit testing
const mocks = new pulumi.runtime.Mocks();

mocks.mockResource = (urn: string, type: string, name: string, id: string) => {
  return {
    id,
    urn,
    state: { name },
  };
};

pulumi.runtime.setMocks(mocks);

describe("VPC Infrastructure", () => {
  it("creates a VPC with correct CIDR", async () => {
    const vpc = new aws.ec2.Vpc("test-vpc", {
      cidrBlock: "10.0.0.0/16",
    });

    const result = await pulumi.all([vpc.cidrBlock]).apply(([cidr]) => cidr);
    expect(result).toBe("10.0.0.0/16");
  });

  it("creates correct number of subnets", async () => {
    const subnets = Array.from({ length: 3 }, (_, i) =>
      new aws.ec2.Subnet(`subnet-${i}`, {
        vpcId: "vpc-123",
        cidrBlock: `10.0.${i}.0/24`,
      })
    );

    expect(subnets).toHaveLength(3);
  });
});
```

## Ansible

### Overview

Ansible = **imperative configuration management** — agentless, chạy qua SSH, dùng YAML playbooks.

```
┌─────────────────────────────────────────────────────────────┐
│  Ansible Flow                                                │
│                                                              │
│  Inventory (hosts)                                           │
│       │                                                      │
│  ansible-playbook -i inventory.ini playbook.yml              │
│       │                                                      │
│  ┌────▼─────────────────────────────────────────┐          │
│  │  Playbook                                    │          │
│  │  ├─ hosts: webservers                        │          │
│  │  ├─ vars: ...                                │          │
│  │  └─ tasks:                                  │          │
│  │       ├─ name: Install nginx                │          │
│  │       ├─ name: Configure nginx              │          │
│  │       └─ name: Start nginx                  │          │
│  └─────────────────────────────────────────────┘           │
│       │                                                      │
│  ┌────▼─────────────────────────────────────────┐          │
│  │  Idempotent Modules                          │          │
│  │  • package: install nginx                   │          │
│  │  • copy: copy config file                   │          │
│  │  • service: start nginx                     │          │
│  │  • command: run custom command              │          │
│  └─────────────────────────────────────────────┘           │
└─────────────────────────────────────────────────────────────┘
```

### Playbook

```yaml
# playbook.yml
---
- name: Deploy MyApp to Production
  hosts: webservers
  become: yes  # sudo
  gather_facts: true

  vars:
    app_version: "1.2.3"
    app_port: 8080
    deploy_user: deploy

  environment:
    HTTPS_PROXY: "{{ http_proxy }}"

  pre_tasks:
    - name: Update apt cache
      ansible.builtin.apt:
        update_cache: yes
        cache_valid_time: 3600
      when: ansible_os_family == "Debian"

    - name: Create deploy user
      ansible.builtin.user:
        name: "{{ deploy_user }}"
        shell: /bin/bash
        groups: docker
        append: yes

  tasks:
    - name: Install Docker
      ansible.builtin.include_role:
        name: geerlingguy.docker
      vars:
        docker_packages_state: present
        docker_install_compose: true

    - name: Create app directory
      ansible.builtin.file:
        path: /opt/myapp
        state: directory
        owner: "{{ deploy_user }}"
        group: "{{ deploy_user }}"
        mode: '0755'

    - name: Deploy application
      ansible.builtin.docker_compose_v2:
        project_src: /opt/myapp
        state: present
      vars:
        version: "{{ app_version }}"

    - name: Configure health check
      ansible.builtin.copy:
        dest: /etc/nginx/conf.d/myapp.conf
        content: |
          upstream myapp_backend {
            server 127.0.0.1:{{ app_port }};
          }
          server {
            listen 80;
            location /healthz {
              return 200 'ok';
            }
          }
      notify: Reload nginx

    - name: Wait for app to be ready
      ansible.builtin.uri:
        url: "http://127.0.0.1:{{ app_port }}/ready"
        status_code: "200"
      register: health_check
      retries: 30
      delay: 5
      until: health_check.status == 200

  handlers:
    - name: Reload nginx
      ansible.builtin.service:
        name: nginx
        state: reloaded
```

## State Management

### Local vs Remote State

```hcl
# Local state (❌ không dùng cho production)
terraform {
  # Không có backend → local
}

# Remote state (✅ production)
terraform {
  backend "s3" {
    bucket = "myorg-terraform-state"
    key    = "production/main/terraform.tfstate"
    region = "ap-southeast-1"
    encrypt = true
    # DynamoDB for state locking (ngăn concurrent apply)
    dynamodb_table = "terraform-state-locks"
  }
}
```

### State Locking (DynamoDB)

```hcl
# State locking ngăn việc 2 người cùng chạy terraform apply
terraform {
  backend "s3" {
    bucket         = "myorg-terraform"
    key            = "prod/terraform.tfstate"
    dynamodb_table = "terraform-locks"  # Bắt buộc cho multi-user
  }
}
```

### State Security

| Risk | Mitigation |
|------|-----------|
| State chứa secrets | `sensitive = true` → ẩn trong output, dùng vault |
| State corruption | Enable versioning trên S3 bucket |
| Concurrent apply | DynamoDB state locking |
| State drift | `terraform plan` = drift detection |
| Lost state | S3 + DynamoDB replicated |

## Terraform vs Pulumi vs Ansible

| | Terraform | Pulumi | Ansible |
|---|---|---|
| **Paradigm** | Declarative | Declarative (imperative possible) | Imperative |
| **Language** | HCL | TypeScript, Python, Go, C# | YAML |
| **State** | `.tfstate` file | `.pulumi/` directory | Stateless (stateless) |
| **Planning** | `terraform plan` | `pulumi preview` | `ansible-playbook --check` |
| **Use case** | Cloud infrastructure | Cloud infrastructure + app | Configuration management, app deploy |
| **Best for** | Multi-cloud provisioning | Complex infra with app logic | Existing servers, app deployment |
| **Agent** | None | None | SSH (agentless) |
| **Secrets** | External (Vault, AWS SM) | Native `pulumi.secret()` | Vault integration |

## Terraform in CI/CD

```yaml
# GitHub Actions
name: Terraform

on:
  push:
    branches: [main]
    paths:
      - 'infra/**'
      - '**.tf'

jobs:
  terraform:
    runs-on: ubuntu-latest
    permissions:
      id-token: write  # Required for OIDC
      contents: read

    steps:
      - uses: actions/checkout@v4

      - name: Configure AWS credentials (OIDC)
        uses: aws-actions/configure-aws-credentials@v4
        with:
          role-to-assume: arn:aws:iam::123456789:role/terraform-deploy
          aws-region: ap-southeast-1

      - name: Setup Terraform
        uses: hashicorp/setup-terraform@v3
        with:
          terraform_version: 1.7.0

      - name: Terraform Init
        working-directory: infra/production
        run: terraform init

      - name: Terraform Format Check
        working-directory: infra/production
        run: terraform fmt -check -recursive

      - name: Terraform Plan
        working-directory: infra/production
        run: terraform plan -input=false -out=tfplan
        env:
          TF_VAR_environment: production

      - name: Terraform Apply
        if: github.ref == 'refs/heads/main'
        working-directory: infra/production
        run: terraform apply -input=false -auto-approve tfplan
```

### Terragrunt (DRY Terraform)

```hcl
# infra/production/terragrunt.hcl
terraform {
  source = "../../modules//eks-cluster"
}

inputs = {
  cluster_name       = "myapp-prod"
  cluster_version    = "1.29"
  vpc_id             = dependency.vpc.outputs.vpc_id
  subnet_ids         = dependency.vpc.outputs.private_subnet_ids
  eks_desired_size   = 3
}

dependencies {
  outputs = ["vpc_id", "private_subnet_ids", "eks_cluster_endpoint"]

  # Load outputs từ module VPC
  resources = [
    "../../vpc/terragrunt.hcl"
  ]
}

# Auto-gen backend config (tránh repeat backend trong mỗi .tf file)
generate "backend" {
  path      = "backend.tf"
  if_exists = "overwrite_terragrunt"
  contents  = <<EOF
terraform {
  backend "s3" {
    bucket = "myorg-terraform"
    key    = "production/eks/terraform.tfstate"
```

## Drift Detection & Remediation

```bash
# Phát hiện drift (infra thực tế ≠ state)
terraform plan -detailed-exitcode
# Exit code 0: no changes, 1: drift detected, 2: error

# Drift remediation
terraform apply   # ✅ Apply state → infra (default)
# hoặc:
terraform import <resource_address> <resource_id>
# Import existing resource vào state

# Import example
terraform import aws_vpc.existing vpc-12345678

# State manipulation (⚠️ dangerous)
terraform state mv aws_instance.old aws_instance.new
terraform state rm aws_instance.deprecated
terraform state pull  # Export state ra JSON
```

## Security Best Practices

```hcl
# 1. S3 backend với versioning + encryption
terraform {
  backend "s3" {
    bucket         = "myorg-terraform-state"
    key            = "prod/terraform.tfstate"
    encrypt        = true
    dynamodb_table = "terraform-locks"
  }
}

# 2. Không hardcode secrets
variable "db_password" {
  sensitive = true
  type      = string
}

# Dùng: terraform apply -var="db_password=xxx"
# Hoặc: AWS Secrets Manager / HashiCorp Vault

# 3. Policy as Code (OPA)
# terraform validate với custom policies
# Checkov cho security scanning

# 4. OIDC (OpenID Connect) — không dùng long-lived credentials
provider "aws" {
  region = "ap-southeast-1"

  # Role assumed via GitHub OIDC
  assume_role {
    role_arn = "arn:aws:iam::123456789:role/github-actions-terraform"
  }
}
```
